#include <iostream>
using namespace std;

//               pozicije/indeksi
//                0  1  2  3  4 
//               {7, 8, 7, 8, 9};         5
float prosjekOcjena(int* ocjene, int brojOcjena) {

	float suma = 0;

	for (int i = 0; i < brojOcjena; i++) // 0 1 2 3 4
	{

		suma += ocjene[i]; // +7 +8 +7 +8 +9 = 39


	}

	return suma / brojOcjena;

}


int brojIznadProsjek(int* ocjene, int brojOcjena) {

	float prosjek = prosjekOcjena(ocjene, brojOcjena); // 7.8

	int brojac = 0;

	for (int i = 0; i < brojOcjena; i++)
	{
		if (ocjene[i] >= prosjek) {

			brojac++;

		}


	}

	return brojac;


}






void main() {

	int BrojOcjena = 5;

	int* OcjeneZaProsjek = new int[BrojOcjena] {7, 8, 7, 8, 9};

	cout << "Ocjene: " << endl;

	for (int i = 0; i < BrojOcjena; i++)
	{
		cout << OcjeneZaProsjek[i] << endl;
	}

	cout << endl;

	float Prosjek = prosjekOcjena(OcjeneZaProsjek, BrojOcjena);

	cout <<"Prosjek tvojih ocjena je: " <<  Prosjek << endl;
	cout <<"Broj ocjena koje su iznad prosjeka je: "<< brojIznadProsjek(OcjeneZaProsjek, BrojOcjena) << endl;



	system("pause>0");

}

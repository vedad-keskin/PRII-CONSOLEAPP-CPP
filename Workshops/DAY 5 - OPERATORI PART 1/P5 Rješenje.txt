#include<iostream>
using namespace std;


// Funkcija za DEEP COPY stringa (jer radimo sa char*)
char* GetNizKaraktera(const char* tekst) {
	if (tekst == nullptr)
		return nullptr;

	int vel = strlen(tekst) + 1; // +1 zbog '\0'
	char* novi = new char[vel];  // alokacija nove memorije
	strcpy_s(novi, vel, tekst);  // kopiranje sadržaja
	return novi; // vraća NOVI string → mora se delete[] kasnije
}


class Artikal {
	char* _sifra;
	char* _naziv;
	double _cijena;
	int _kolicinaNaSkladistu;
public:

	Artikal(const char* sifra = "SIF000", const char* naziv = "---", double cijena = 0.0, int kolicina = 0) {

		// svaki string se kopira → nema dijeljenja memorije
		_sifra = GetNizKaraktera(sifra);
		_naziv = GetNizKaraktera(naziv);

		_cijena = cijena;
		_kolicinaNaSkladistu = kolicina;
	}

	Artikal(const Artikal& obj) {

		// COPY KONSTRUKTOR → deep copy
		_sifra = GetNizKaraktera(obj._sifra);
		_naziv = GetNizKaraktera(obj._naziv);

		_cijena = obj._cijena;
		_kolicinaNaSkladistu = obj._kolicinaNaSkladistu;
	}

	Artikal& operator = (const Artikal& obj) {
		if (this != &obj) {

			// brišemo staru memoriju
			delete[] _sifra;
			delete[] _naziv;

			// pravimo novu kopiju
			_sifra = GetNizKaraktera(obj._sifra);
			_naziv = GetNizKaraktera(obj._naziv);

			_cijena = obj._cijena;
			_kolicinaNaSkladistu = obj._kolicinaNaSkladistu;
		}
		return *this;
	}

	~Artikal() {

		// oslobađanje memorije → OBAVEZNO kod char*
		delete[] _sifra;
		_sifra = nullptr;
		delete[] _naziv;
		_naziv = nullptr;
	}

	friend ostream& operator << (ostream& COUT, const Artikal& obj) {

		COUT << obj._naziv << " [" << obj._sifra << "] - " << obj._cijena << " KM" << endl;
		COUT << "Kolicina na skladistu: " << obj._kolicinaNaSkladistu << " kom" << endl;

		return COUT;
	}


	// Operator -= na način da se stanje na skladištu artikla promijeni za proslijeđenu vrijednost

	//artikal1 -= 10; // op -= 

	// ↑ OVA LINIJA:
	// koristi operator -= kao skraćeni zapis za smanjenje količine

	Artikal& operator -= (int kolicina) {

		if (kolicina > _kolicinaNaSkladistu) {
			// zaštita → ne može ispod 0
			return *this;
		}

		//_kolicinaNaSkladistu -= kolicina;
		// ↑ kraći zapis

		_kolicinaNaSkladistu = _kolicinaNaSkladistu - kolicina;
		// ↑ duži zapis (isti rezultat)

		return *this;
	}


	// Operator += na način da se stanje na skladištu artikla promijeni za proslijeđenu vrijednost

	Artikal& operator += (int kolicina) {

		_kolicinaNaSkladistu += kolicina;
		// ↑ skraćeni zapis

		//_kolicinaNaSkladistu = _kolicinaNaSkladistu + kolicina;
		// ↑ isti rezultat, samo duže napisano

		return *this;
	}

	// Operator ++ na način da se cijena artikla uveća za 1

	//artikal1++; // op ++

	// ↑ koristi postfix operator (ali ti si implementirao prefiksni!)

	// ++ i -- su operatori -> prefiksni op 

	Artikal& operator ++ () {

		_cijena++;
		// ↑ najkraći zapis

		//_cijena += 1;
		// ↑ isto

		//_cijena = _cijena + 1;
		// ↑ isto, ali najduže

		return *this;
	}


	// Operator -- na način da se cijena artikla umanji za 1

	Artikal& operator -- () {

		if (_cijena < 1) {
			// zaštita → ne ide u minus
			return *this;
		}

		_cijena--;
		return *this;
	}


	// Operator == koji poredi artikle na osnovu sifre

	// bool --> operatori poredjena --> == != > < >= <= 

	//if (artikal1 == artikal2) {

	// ↑ koristi operator ==

	bool operator == (const Artikal& obj) {

		//if (strcmp(_sifra , obj._sifra) == 0) {
		//	return true;
		//}
		//else {
		//	return false;
		//}
		// ↑ klasični if-else način

		return strcmp(_sifra, obj._sifra) == 0;
		// ↑ skraćeni način

		// ? : 
		//return strcmp(_sifra, obj._sifra) == 0 ? true : false;
		// ↑ ternarni operator (nije potreban jer već vraća bool)
	}

	// READ-ONLY NACIN PRISTUPA GLAVNIM PAR.

	int GetKolicinaNaSkladistu() { return _kolicinaNaSkladistu; }
	double GetCijena() { return _cijena; }
	char* GetSifra() { return _sifra; }
	char* GetNaziv() { return _naziv; }
};


class StavkaRacuna {
	Artikal* _artikal; // ( ) → pointer, mora se new/delete
	int* _kolicina;    // ( ) → pointer na int
public:

	StavkaRacuna(Artikal artikal = Artikal(), int kolicina = 0) {

		_artikal = new Artikal(artikal); // kopija objekta
		_kolicina = new int(kolicina);   // alokacija int-a
	}

	StavkaRacuna(const StavkaRacuna& obj) {

		_artikal = new Artikal(*obj._artikal);
		_kolicina = new int(*obj._kolicina);
	}

	StavkaRacuna& operator = (const StavkaRacuna& obj) {
		if (this != &obj) {

			delete _artikal;
			delete _kolicina;

			_artikal = new Artikal(*obj._artikal);
			_kolicina = new int(*obj._kolicina);
		}
		return *this;
	}

	~StavkaRacuna() {

		delete _artikal;
		_artikal = nullptr;
		delete _kolicina;
		_kolicina = nullptr;
	}

	friend ostream& operator << (ostream& COUT, const StavkaRacuna& obj) {
		COUT << *obj._artikal;
		COUT << "Kupljena kolicina: " << *obj._kolicina << " kom" << endl << endl;
		return COUT;
	}

	// Operator += koji uvećava stanje na skladištu artikla

	//stavka1 += 10;

	StavkaRacuna& operator += (int kolicina) {

		// ❌ NE RADI → getter vraća vrijednost, ne referencu

		//_artikal._kolicinaNaSkladistu += kolicina;
		// ❌ NE RADI → StavkaRacuna nema pristup privatnom parametru _kolicinaNaSkladistu

		_artikal->operator+=(kolicina);
		// ✅ ISPRAVNO → poziva operator iz Artikal klase

		_artikal->operator-=(kolicina);

		return *this;

		//return strcmp(_artikal->GetSifra(), obj._artikal->GetSifra()) == 0;

		// pravi NOVU memoriju svaki put

		// nepotrebno komplikovano

		return *this;
	}

		// 1.20 * 3 = 3.60
		return _artikal->GetCijena() * *_kolicina;
	}

	double GetKolicina() {
		return *_kolicina;
	}
};


class Racun {
	static int _id; // globalni brojač računa
	int _brojRacuna;

	StavkaRacuna* _stavke;
	int _trenutno;
	int _max;

	double _iznosBezPDV;
	double _iznosSaPDV;

	bool _naplaceno;
public:

	Racun(int max = 3) {

		// AKO JE 0 NA POCETKU

		_id++;
		_brojRacuna = _id;

		//_brojRacuna = ++_id; 
		// ↑ isti rezultat (pre-increment)

		// AKO JE 1 NA POCETKU

		//_brojRacuna = _id;
		//_id++;
		// ↑ druga logika (razlika u startu)

		//_brojRacuna = _id++;
		// ↑ post-increment (uzima staru vrijednost pa povećava)

		_trenutno = 0;
		_max = max;

		_stavke = new StavkaRacuna[_max];

		_iznosBezPDV = 0;
		_iznosSaPDV = 0;

		_naplaceno = false;
	}

	Racun(const Racun& obj) {

		//_id = obj._id; STATIC se NIKAD NE KOPIRA

		_brojRacuna = obj._brojRacuna;

		_trenutno = obj._trenutno;
		_max = obj._max;

		_stavke = new StavkaRacuna[_max];

		for (int i = 0; i < _trenutno; i++)
			_stavke[i] = obj._stavke[i];

		_iznosBezPDV = obj._iznosBezPDV;
		_iznosSaPDV = obj._iznosSaPDV;

		_naplaceno = obj._naplaceno;
	}

	Racun& operator = (const Racun& obj) {
		if (this != &obj) {

			delete[] _stavke;

			_brojRacuna = obj._brojRacuna;
			_trenutno = obj._trenutno;
			_max = obj._max;

			_stavke = new StavkaRacuna[_max];

			for (int i = 0; i < _trenutno; i++)
				_stavke[i] = obj._stavke[i];

			_iznosBezPDV = obj._iznosBezPDV;
			_iznosSaPDV = obj._iznosSaPDV;

			_naplaceno = obj._naplaceno;
		}
		return *this;
	}


	~Racun() {
		delete[] _stavke;
		_stavke = nullptr;
	}


	// Operator () → ponaša se kao funkcija
	Racun& operator () (Artikal artikal, int kolicina) {

		if (_max == _trenutno)
			return *this;

		StavkaRacuna stavka(artikal, kolicina);

		stavka.operator-=(kolicina);
		// ↓ znači: kupovina → smanjuje stanje artikla

		for (int i = 0; i < _trenutno; i++)
		{
			if (_stavke[i] == stavka)
				return *this; // duplikat → ignoriše
		}

		// iznos BEZ +=   1.20 * 3 .. 
		_iznosBezPDV += stavka.GetCijena();

		// iznos SA                 3.60 + 18% od 3.60
		//_iznosSaPDV += stavka.GetCijena() + stavka.GetCijena() * 0.18;
		// ↑ klasični zapis

		_iznosSaPDV += stavka.GetCijena() * 1.18;
		// ↑ kraći zapis

		_stavke[_trenutno] = stavka;
		_trenutno++;

		return *this;
	}


	void PrintRacun() {

		//if (_naplaceno == true)
		// ↑ isto kao dolje

		if (_naplaceno) {
			cout << "Racun je vec naplacen" << endl;
			return;
		}

		cout << "======= Br. racuna " << _brojRacuna << " =========" << endl << endl;

		for (int i = 0; i < _trenutno; i++)
		{
			cout << _stavke[i];
			cout << "===================" << endl << endl;
		}

		cout << "Iznos bez PDV: " << _iznosBezPDV << " KM" << endl;
		cout << "Iznos sa PDV: " << _iznosSaPDV << " KM" << endl;

		_naplaceno = true;
	}
};

/*
Funkcija GenerisiOznaku generise oznaku posiljke u formatu: BH-VD-BBBB-IN
Potpis funkcije treba biti:
string GenerisiOznaku(const char* imePrezime, int redniBroj, VrstaDostave vrstaDostave)
BH -> fiksna oznaka drzave,
VD -> vrsta dostave: ST (standardna), BR (brza), MD (medjunarodna),
BBBB -> redni broj posiljke popunjen nulama na slobodnim mjestima,
IN -> inicijali imena i prezimena primaoca.
Kod imena koje sadrzi vise rijeci koriste se inicijal prve i posljednje
rijeci. Redni broj mora biti u rasponu 1-9999.
Za neispravne podatke funkcija vraca "BH-XX-0000-XX".
*/

string GenerisiOznaku(string imeprezime, int id, VrstaDostave vd) {


	string oznaka;

	// BH -> fiksna oznaka drzave

	oznaka += "BH-";

	// VD -> vrsta dostave: ST (standardna), BR (brza), MD (medjunarodna)

	oznaka += VrstaDostaveOznake[vd];

	oznaka += "-";

	// BBBB -> redni broj posiljke popunjen nulama na slobodnim mjestima

	stringstream ss;

	ss << setfill('0') << setw(4) << id;

	oznaka += ss.str();

	oznaka += "-";

	// IN->inicijali imena i prezimena primaoca.

	char prvoSlovoImena = imeprezime[0];
	char prvoSlovoPrezimena = imeprezime[imeprezime.rfind(' ') + 1];

	oznaka += prvoSlovoImena;
	oznaka += prvoSlovoPrezimena;


	return oznaka;

}


/*
Koristeci regex, funkcija ValidirajOznaku provjerava da li je oznaka
zapisana u prethodno definisanom formatu. Dozvoljene su iskljucivo oznake
vrsta ST, BR i MD, redni broj mora imati cetiri cifre i ne moze biti 0000,
a inicijali moraju biti zapisani velikim slovima.
Potpis funkcije treba biti:
bool ValidirajOznaku(const string& oznaka)
*/

// BH-VD-BBBB-IN

bool ValidirajOznaku(const string& oznaka) {

	regex pattern("(BH-)(ST|BR|MD)[-](?!0000)[0-9]{4}[-][A-Z]{2}");

	return regex_match(oznaka, pattern);

}




template<class T1, class T2, int max>
class Kolekcija {
	T1* _elementi1[max];
	T2* _elementi2[max];
	int _trenutno;
public:

	Kolekcija() : _trenutno(0) {

		for (int i = 0; i < max; i++) {
			_elementi1[i] = nullptr;
			_elementi2[i] = nullptr;
		}

	}

	Kolekcija(const Kolekcija& obj) {

		_trenutno = obj._trenutno;

		for (int i = 0; i < _trenutno; i++)
		{
			_elementi1[i] = new T1(*obj._elementi1[i]);
			_elementi2[i] = new T2(*obj._elementi2[i]);
		}

	}

	Kolekcija& operator = (const Kolekcija& obj) {
		if (this != &obj) {

			for (int i = 0; i < _trenutno; i++) {
				delete _elementi1[i];
				delete _elementi2[i];
			}
			_trenutno = 0;

			_trenutno = obj._trenutno;

			for (int i = 0; i < _trenutno; i++)
			{
				_elementi1[i] = new T1(*obj._elementi1[i]);
				_elementi2[i] = new T2(*obj._elementi2[i]);
			}
		}
		return *this;
	}

	void Dodaj(T1 el1, T2 el2) {

		if (max == _trenutno) {
			return;
		}

		_elementi1[_trenutno] = new T1(el1);
		_elementi2[_trenutno] = new T2(el2);

		_trenutno++;

	}

	/*
	DodajNaPoziciju dodaje novi par na lokaciju/indeks definisanu prvim parametrom,
	pomjera postojece pokazivace udesno i vraca trenutno stanje
	kolekcije (izmijenjeni tj. prosireni objekat).
	U slucaju popunjene kolekcije ili neispravne lokacije potrebno je baciti izuzetak.
	*/

	Kolekcija DodajNaPoziciju(int lokacija, T1 el1, T2 el2) {

		if (lokacija < 0 || lokacija > _trenutno || max == _trenutno) {
			throw exception("Lokacija ne postoji ili je kolekcija popunjena");
		}

		Kolekcija prosireniInventar;

		for (int i = 0; i < _trenutno; i++)
		{
			if (i == lokacija) {
				prosireniInventar.Dodaj(el1, el2);
			}
			prosireniInventar.Dodaj(*_elementi1[i], *_elementi2[i]);
		}

		if (lokacija == _trenutno) {
			prosireniInventar.Dodaj(el1, el2);
		}

		*this = prosireniInventar;


		return *this;
	}

	/*
	UkloniSaPozicije uklanja par sa proslijedjene pozicije/indeksa,
	preostale elemente pomjera ulijevo i vraca uklonjeni par u obliku pair<T1, T2>.
	Za poziciju izvan opsega baca se izuzetak.
	*/
	//pair<int, string> uklonjeni = prosireniBrojevi.UkloniSaPozicije(2);

	pair<int, string> UkloniSaPozicije(int pozicija) {

		if (pozicija < 0 || pozicija >= _trenutno) {

			throw exception("Lokacija ne postoji");
		}


		pair<int, string> uklonjeni(*_elementi1[pozicija], *_elementi2[pozicija]);

		Kolekcija prosireniBrojevi;

		for (int i = 0; i < _trenutno; i++)
		{

			if (pozicija != i) {

				prosireniBrojevi.Dodaj(*_elementi1[i], *_elementi2[i]);

			}

		}

		*this = prosireniBrojevi;


		return uklonjeni;

	}



	int GetTrenutno() const { return _trenutno; }
	T1& GetElement1(int indeks) { return *_elementi1[indeks]; }
	T2& GetElement2(int indeks) { return *_elementi2[indeks]; }
	const T1& GetElement1(int indeks) const { return *_elementi1[indeks]; }
	const T2& GetElement2(int indeks) const { return *_elementi2[indeks]; }
	T1& operator[](int indeks) { return *_elementi1[indeks]; }

	friend ostream& operator<<(ostream& COUT, const Kolekcija& obj) {
		for (int i = 0; i < obj.GetTrenutno(); i++)
			COUT << obj.GetElement1(i) << " " << obj.GetElement2(i) << endl;
		return COUT;
	}

	~Kolekcija() {
		for (int i = 0; i < _trenutno; i++) {
			delete _elementi1[i];
			delete _elementi2[i];
			_elementi1[i] = nullptr;
			_elementi2[i] = nullptr;
		}
		_trenutno = 0;
	}
};
class DatumVrijeme {
	int* _dan, * _mjesec, * _godina, * _sati, * _minute;
public:
	DatumVrijeme(int dan = 1, int mjesec = 1, int godina = 2000, int sati = 0, int minute = 0) {
		_dan = new int(dan);
		_mjesec = new int(mjesec);
		_godina = new int(godina);
		_sati = new int(sati);
		_minute = new int(minute);
	}

	DatumVrijeme(const DatumVrijeme& obj) {

		_dan = new int(*obj._dan);
		_mjesec = new int(*obj._mjesec); 
		_godina = new int(*obj._godina);
		_sati = new int(*obj._sati);
		_minute = new int(*obj._minute);
	}

	DatumVrijeme& operator = (const DatumVrijeme& obj) {
		if (this != &obj) {

			delete _dan;
			delete _mjesec;
			delete _godina;
			delete _sati;
			delete _minute;

			_dan = new int(*obj._dan);
			_mjesec = new int(*obj._mjesec); 
			_godina = new int(*obj._godina);
			_sati = new int(*obj._sati);
			_minute = new int(*obj._minute);
		}
		return *this;
	}

	bool operator == (const DatumVrijeme& obj) {
         return *_dan == *obj._dan && *_mjesec == *obj._mjesec && *_godina == *obj._godina && *_sati == *obj._sati && *_minute == *obj._minute;
	}

	int GetSumaDana() const {

		return *_dan + *_mjesec * 30 + *_godina * 365;

	}

	int GetSumaMinuta() const {
		return *_minute + *_sati * 60 + GetSumaDana() * 24 * 60;
	}

	bool operator > (const DatumVrijeme& obj) const {
		return GetSumaMinuta() > obj.GetSumaMinuta();
	}
	
	string ToString() const {

		stringstream ss;

		ss << setfill('0') << setw(2) << *_dan << "." << setw(2) << *_mjesec << "." << *_godina << " " << setw(2) << *_sati << ":" << setw(2) << *_minute;

		return ss.str();	

	}

	friend ostream& operator << (ostream& COUT, const DatumVrijeme& obj)
	{
		//ToString vraca datum i vrijeme u formatu DD.MM.GGGG HH:MM,
		COUT << obj.ToString();
		return COUT;
	}

	~DatumVrijeme() {
		delete _dan;
		delete _mjesec;
		delete _godina;
		delete _sati;
		delete _minute;
		_dan = _mjesec = _godina = _sati = _minute = nullptr;
	}
};
class Posiljka {
	char* _oznaka;
	char* _sadrzaj;
	char* _imePrezimePrimaoca;
	char* _emailPrimaoca;
	VrstaDostave _vrstaDostave;
	int _redniBroj;
	Kolekcija<StatusPosiljke, DatumVrijeme, 10> _statusi;
	void Dealociraj() {
		delete[] _oznaka;
		delete[] _sadrzaj;
		delete[] _imePrezimePrimaoca;
		delete[] _emailPrimaoca;
		_oznaka = _sadrzaj = _imePrezimePrimaoca = _emailPrimaoca = nullptr;
	}
public:
	const char* GetOznaka() const { return _oznaka; }
	const char* GetSadrzaj() const { return _sadrzaj; }
	const char* GetImePrezimePrimaoca() const { return _imePrezimePrimaoca; }
	const char* GetEmailPrimaoca() const { return _emailPrimaoca; }
	VrstaDostave GetVrstaDostave() const { return _vrstaDostave; }
	int GetRedniBroj() const { return _redniBroj; }
	Kolekcija<StatusPosiljke, DatumVrijeme, 10>& GetStatusi() { return _statusi; }
	StatusPosiljke GetTrenutniStatus()const {
		return _statusi.GetElement1(_statusi.GetTrenutno() - 1);
	}

	/*
	Konstruktor Posiljke generise oznaku i evidentira pocetni status KREIRANA
	sa proslijedjenim vremenom kreiranja.
	*/
	//Posiljka dokumenti("Dokumenti", "Amina Buric", "amina@fit.ba", BRZA, 42, kreirana);

	Posiljka(const char* sadrzaj, const char* imeprezime , const char* email, VrstaDostave vrsta, int rednibroj, DatumVrijeme datumvrijeme  ) {

		_sadrzaj = AlocirajTekst(sadrzaj);
		_imePrezimePrimaoca = AlocirajTekst(imeprezime);
		_emailPrimaoca = AlocirajTekst(email);

		_vrstaDostave = vrsta;
		_redniBroj = rednibroj;

		_oznaka = AlocirajTekst(GenerisiOznaku(imeprezime,rednibroj,vrsta).c_str());

	
		_statusi.Dodaj(KREIRANA, datumvrijeme);

	}

	Posiljka(const Posiljka& obj) {

		_sadrzaj = AlocirajTekst(obj._sadrzaj);
		_imePrezimePrimaoca = AlocirajTekst(obj._imePrezimePrimaoca);
		_emailPrimaoca = AlocirajTekst(obj._emailPrimaoca);
		_oznaka = AlocirajTekst(obj._oznaka);

		_vrstaDostave = obj._vrstaDostave;
		_redniBroj = obj._redniBroj;

		_statusi = obj._statusi;

	}

	Posiljka& operator = (const Posiljka& obj) {
		if (this != &obj) {

			Dealociraj();


			_sadrzaj = AlocirajTekst(obj._sadrzaj);
			_imePrezimePrimaoca = AlocirajTekst(obj._imePrezimePrimaoca);
			_emailPrimaoca = AlocirajTekst(obj._emailPrimaoca);
			_oznaka = AlocirajTekst(obj._oznaka);

			_vrstaDostave = obj._vrstaDostave;
			_redniBroj = obj._redniBroj;

			_statusi = obj._statusi;
		}
		return *this;
	}

	bool operator == (const Posiljka& obj) const {

		return _redniBroj == obj._redniBroj || strcmp(_oznaka, obj._oznaka) == 0;

	}

	/*
	DodajStatus dodaje status samo ako je vrijeme vece od vremena
	posljednjeg statusa i ako je promjena statusa dozvoljena.
	Dozvoljen slijed statusa je: KREIRANA -> PREUZETA -> U_TRANZITU -> ISPORUCENA
	OTKAZANA se moze evidentirati iz bilo kojeg statusa koji nije zavrsni.
	Nakon statusa ISPORUCENA ili OTKAZANA nisu dozvoljene nove promjene.
	Preskakanje, ponavljanje statusa i hronoloski neispravno vrijeme vracaju
	false, bez izmjene kolekcije statusa.
	*/
	bool DodajStatus(StatusPosiljke novistatus, DatumVrijeme datumvrijeme) {

		// 1) Trenutni status i vrijeme
		StatusPosiljke trenutniStatus = GetTrenutniStatus();
		DatumVrijeme trenutnoDatumVrijeme = _statusi.GetElement2(_statusi.GetTrenutno() - 1);

		// 2) Novo vrijeme mora biti vece od starog zadnjeg
		if (!(datumvrijeme > trenutnoDatumVrijeme)) {
			return false;
		}

		// 3) Ako je stanje zavrseno (ISPORUCENA ili OTKAZANA) -> Nema vise promjena
		if (trenutniStatus == ISPORUCENA || trenutniStatus == OTKAZANA) {
			return false;
		}

		// 4) OTKAZANA se smije dodati iz bilo kojeg NEZAVRSNOG statusa
		if (novistatus == OTKAZANA) {
			_statusi.Dodaj(novistatus, datumvrijeme);
			return true;
		}

		// 5) inace mora ici redom, bez preskakanja i bez ponavljanja:
		//    KREIRANA -> PREUZETA -> U_TRANZITU -> ISPORUCENA

		if (trenutniStatus == KREIRANA && novistatus == PREUZETA) {

			_statusi.Dodaj(novistatus, datumvrijeme);
			return true;

		}
		else if (trenutniStatus == PREUZETA && novistatus == U_TRANZITU) {

			_statusi.Dodaj(novistatus, datumvrijeme);
			return true;

		}
		else if (trenutniStatus == U_TRANZITU && novistatus == ISPORUCENA) {

			_statusi.Dodaj(novistatus, datumvrijeme);
			return true;

		}
		else {

			return false;
		}


	}

	string ToString() const {
		stringstream ss;
		ss << _oznaka << " | " << _imePrezimePrimaoca << " | " << _vrstaDostave << " | " << _sadrzaj << " | " << GetTrenutniStatus();
		return ss.str();
	}


	friend ostream& operator<<(ostream& COUT, const Posiljka& obj) {
		// ToString vraca podatke u formatu:
		// oznaka | ime i prezime | vrsta dostave | sadrzaj | trenutni status
		// BH-BR-0042-AB | Amina Buric | BRZA | Dokumenti | KREIRANA
		COUT << obj.ToString();
		return COUT;
	}
	~Posiljka() {
		Dealociraj();
	}
};
class DostavnaSluzba {
	char* _naziv;
	vector<Posiljka> _posiljke;
public:
	DostavnaSluzba(const char* naziv = "") {
		_naziv = AlocirajTekst(naziv);
	}

	DostavnaSluzba(const DostavnaSluzba& obj) {

		_naziv = AlocirajTekst(obj._naziv);
		_posiljke = obj._posiljke;

	}

	DostavnaSluzba& operator = (const DostavnaSluzba& obj) {
		if (this != &obj) {

			delete[] _naziv;


			_naziv = AlocirajTekst(obj._naziv);
			_posiljke = obj._posiljke;

		}
		return *this;
	}

	bool operator == (const DostavnaSluzba& obj) {
		return strcmp(_naziv, obj._naziv) == 0;
	}

	void DodajPosiljku(Posiljka posiljka) {


		for (int i = 0; i < _posiljke.size(); i++)
		{
			if (posiljka == _posiljke[i]) {

				throw exception("Posiljka je vec dodana");
			}
		}


		_posiljke.push_back(posiljka);

	}

	const char* GetNaziv() const { return _naziv; }
	vector<Posiljka>& GetPosiljke() { return _posiljke; }

	~DostavnaSluzba() {
		delete[] _naziv;
		_naziv = nullptr;
	}


	friend ostream& operator << (ostream& COUT, const DostavnaSluzba& obj) {
		COUT << "Posta: "<<obj._naziv << endl;

		for (int i = 0; i < obj._posiljke.size(); i++)
		{
			COUT << " - " << obj._posiljke[i] << endl;
		}

		return COUT;
	}

};

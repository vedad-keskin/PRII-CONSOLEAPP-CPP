#include <iostream>
using namespace std;

// POKAZIVACI :  

// 1. ALOKACIJA ( new )  
// ako je jedna vrijednost --> ( )  
//    npr: int* x = new int(5);  --> alocira 1 int i postavlja ga na 5
// ako je vise vrijednosti --> [ ]  
//    npr: char* niz = new char[10]; --> alocira niz od 10 char-ova (NEINICIJALIZOVAN)

// 2. DEREFERENCIRANJE ( ispis , kopiranje )
// ako je jedna vrijednost --> ( ) 
//    _ECTS -> adresa (npr 0x1234)
//    *_ECTS -> stvarna vrijednost na toj adresi (npr 7)

// ako je vise vrijednosti --> [ ] 
//    _naziv -> ispisuje vrijednosti
//    *_naziv -> prvi karakter (npr 'N')

// VAŽNO: niz (char*) se ponaša kao pointer na prvi element!

// 3. DEALOCIRANJE / DESTRUKTOR  --> ( delete , nullptr )
// ako je jedna vrijednost --> ( )
//    delete _ECTS;  --> briše JEDAN int

// ako je vise vrijednosti --> [ ]
//    delete[] _naziv;  --> briše CIJELI niz

//delete:
//
//oslobađa(briše) memoriju koju si ranije zauzeo sa new
//vraća tu memoriju sistemu(heap - u)
//NE briše pointer varijablu, nego ono na šta pokazuje

// OBAVEZNO: koristiti isti oblik kao kod new!
// new -> delete
// new[] -> delete[]

// nullptr:
// nakon brisanja pointer treba postaviti na nullptr
// da izbjegnemo "dangling pointer" (pokazivač koji pokazuje na obrisanu memoriju)

//pointer i dalje postoji
//ali memorija na koju pokazuje više nije validna
//
//➡️ Ovo se zove dangling pointer

// nullptr
//znači "ne pokazujem ni na šta"
//to je specijalna vrijednost za pointere

//new → zauzima memoriju
//delete → briše memoriju
//nullptr → resetuje pointer



// RAZLIKE struct i class

// struct (sve po defaultu public)
// class (sve po default private)

// mogucnosti iste (100% iste u C++)

// razlika je SAMO u default pristupu:
// struct Struktura { int broj; }; --> public
// class Klasa { int broj; };      --> private

// class-e se koriste za kompleksnije stvari dok struct za jednostavnije

// HISTORIJA:
// C already had struct before C++ existed.
// When Bjarne Stroustrup created C++, he kept struct for compatibility with C programs.

// PRAKSA:
// struct → POD (plain data), bez puno logike
// class  → objekti sa funkcijama, enkapsulacija, logika


class Predmet {
public:             //      dft      //     user def.    //
	char* _naziv;  	// Nema predmeta // Programiranje II //
	int _ocjena;	//      5		 //       9			 //
	int* _ECTS;		//      0		 //       7			 //


	// dft. constructor

	Predmet() {

		_ocjena = 5;

		_ECTS = new int(0);

		//char tekst1[] = "Nema Predmeta";
		//string tekst2 = "Nema predmeta";
		const char* dftNaziv = "Nema predmeta"; // string literal --> read only tekst

		// strlen + 1:
		// +1 je zbog '\0' (null terminator) koji označava kraj stringa

		int vel = strlen(dftNaziv) + 1; // 14 

		_naziv = new char[vel]; // na pocetku je taj niz prazan 

		// strcpy_s:
		// sigurnija verzija strcpy (provjerava veličinu)

		// 1. gdje hocete da pohranite nesto 
		// 2. koliko ce karaktera biti prekopirano 
		// 3. sta zelimo da pohranimo
		strcpy_s(_naziv, vel, dftNaziv);


	}


	// user-def constr. , standardni/obicni constr.

	//	Predmet predmet1("Programiranje II", 9, 7);
	Predmet(const char* naziv, int ocjena, int ects) {

		_ocjena = ocjena;

		_ECTS = new int(ects);


		int vel = strlen(naziv) + 1;

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, naziv); // 1. gdje, 2. koliko, 3. sta


	}

	// BEZ ovoga bi se desio SHALLOW COPY:
	// kopirale bi se samo adrese → oba objekta bi koristila ISTU memoriju
	
	// problem:
	// kada jedan objekat se uništi → drugi ostaje sa "pokvarenim" pointerom
	
	// ZATO radimo DEEP COPY:
	// alociramo novu memoriju i kopiramo vrijednosti

	// copy constr. 

	// 	Predmet predmetKopija(predmet1);

	// bez & -> pravimo kopiju ali isto tako pravimo i duplu memoriju
	// sa & -> saljemo original ali se taj original moze mijenjati
	// const -> osigurava da se nesto ne smije mijenjati 

	Predmet(const Predmet& predmet) { // salje originalni predmet

		_ocjena = predmet._ocjena;

		_ECTS = new int(*predmet._ECTS);

		int vel = strlen(predmet._naziv) + 1;

		_naziv = new char[vel];

		strcpy_s(_naziv, vel, predmet._naziv);

	}

	// if (this != &predmet)
	// provjera da li objekat dodjeljuje sam sebi
	
	// bez ovoga:
	// predmet1 = predmet1;
	// prvo bi obrisao svoju memoriju → pa pokušao kopirati iz OBRISANE memorije
	
	// delete prije new:
	// sprječava memory leak (curenje memorije)
	
	// na kraju:
	// return *this;
	// omogućava lančano dodjeljivanje:
	// a = b = c;

	//	predmet5 = predmet1; // <-- operator dodjele

	// & --> salje se original tako sto se posalje adresa objekta i onda se poredi sa this (adresom glavnog objekta)
	// const --> originalni objekat je konstantan i ne moze se mijenjati
	// vraca se *this ---> dereferencirani objekat

	Predmet& operator = (const Predmet& predmet) {

		if (this != &predmet) {


			delete _ECTS;
			delete[] _naziv;


			_ocjena = predmet._ocjena;

			_ECTS = new int(*predmet._ECTS);

			int vel = strlen(predmet._naziv) + 1;

			_naziv = new char[vel];

			strcpy_s(_naziv, vel, predmet._naziv);

		}


		return *this;
	}

	// automatski se poziva kada objekat izlazi iz scope-a

	// npr:
	// {
	//    Predmet p;
	// } --> ovdje se poziva ~Predmet()
	
	// VAŽNO:
	// ako koristiš new → MORAŠ imati destruktor!
	
	// nullptr nakon delete:
	// sigurnosna mjera (nije obavezno ali dobra praksa)

	// destructor ili dealokacija
	// 
	// TILDA -> alt gr + 1
	~Predmet() {

		delete _ECTS;
		_ECTS = nullptr;

		delete[] _naziv;
		_naziv = nullptr;

	}

	// pr1 nacin ispisa

	void Ispis() {

		cout << "Naziv predmeta: " << _naziv << endl;
		cout << "Ocjena predmeta: " << _ocjena << endl;
		cout << "ECTS predmeta: " << *_ECTS << endl;

		cout << endl << "====================" << endl << endl;

	}





};





class Student {
	char* _imePrezime;   // niz karaktera [ ] char

	Predmet* _polozeni;  // [ ] niz Predmeta
	int _trenutno;       // koliko trenutno imamo pohranjenih predmeta u nizu
	int _max;            // koliko maksimalno mozete imati pohranjenih predmeta
public:


	Student() {

		const char* dftImePrezime = "Nema imena i prezime";
		int vel = strlen(dftImePrezime) + 1;
		_imePrezime = new char[vel];
		strcpy_s(_imePrezime, vel, dftImePrezime);

		_trenutno = 0;
		_max = 3;

		_polozeni = new Predmet[_max];


	}


	// 	Student student("Vedad Keskin", 3);
	Student(const char* imeprezime, int max) {

		int vel = strlen(imeprezime) + 1;
		_imePrezime = new char[vel];
		strcpy_s(_imePrezime, vel, imeprezime);


		_trenutno = 0;
		_max = max;

		_polozeni = new Predmet[_max];



	}

	Student(const Student& student) {

		int vel = strlen(student._imePrezime) + 1;
		_imePrezime = new char[vel];
		strcpy_s(_imePrezime, vel, student._imePrezime);


		_trenutno = student._trenutno;

		_max = student._max;

		_polozeni = new Predmet[_max];

		// ZAŠTO petlja?

		for (int i = 0; i < _trenutno; i++)
		{
			_polozeni[i] = student._polozeni[i];
		}

		// jer kopiramo svaki objekat pojedinačno

		// koristi se operator= iz klase Predmet



	}

	// 	studentOperatorDodjele = student;

	Student& operator = (const Student& student) {

		if (this != &student) {

			delete[] _imePrezime;
			delete[] _polozeni;


			int vel = strlen(student._imePrezime) + 1;
			_imePrezime = new char[vel];
			strcpy_s(_imePrezime, vel, student._imePrezime);


			_trenutno = student._trenutno;

			_max = student._max;

			_polozeni = new Predmet[_max];

			for (int i = 0; i < _trenutno; i++) // 0 1 
			{
				_polozeni[i] = student._polozeni[i];
			}

		}

		return *this;

	}



	~Student() {

		delete[] _imePrezime;
		_imePrezime = nullptr;
		delete[] _polozeni;
		_polozeni = nullptr;

	}




	void Ispis() {

		cout << "Ime i prezime studenta: " << _imePrezime << endl;
		cout << "=========== POLOZENI PREDMETI =============" << endl << endl;

		for (int i = 0; i < _trenutno; i++) // 0 1 2
		{
			_polozeni[i].Ispis(); // pol[0] , pol[1] , pol[2] 
		}


	}


	// 	student.AddPolozeniPredmet(predmet1); -> MATII
	void AddPolozeniPredmet(Predmet predmet) {
		// Voditi računa da se ne može prekoračiti kapacitet niza

		if (_trenutno == _max) {
			cout << "Student ne moze imati vise polozenih predmeta" << endl;
			return;
		}


		// Voditi računa da se ne može dodavati isti predmet (istog naziva)

		for (int i = 0; i < _trenutno; i++) // 0 1 
		{
			// Poređenje dva niza karaktera - char*
			// Nije moguće sa char* == char*
			// Mora
			// strcmp( naziv1 , naziv2 ) == 0 

			// ZAŠTO ne možemo koristiti == ?
			// jer char* poredi ADRESE, ne sadržaj!

			//_polozeni[1]._naziv ───► "Matematika II" (adresa 0x100)
			//predmet._naziv ────────► "Matematika II" (adresa 0x200)

			//if (_polozeni[i]._naziv == predmet._naziv) {
			if (strcmp(_polozeni[i]._naziv, predmet._naziv) == 0) {


				cout << "Student vec ima odabrani polozeni predmet" << endl;
				return;

			}


		}

		// Dodaje novi polozeni predmet u niz

		// OVDJE SE POZIVA operator= iz klase Predmet!

		_polozeni[_trenutno] = predmet; // pol[0] = PRII , pol[1] = MATII , pol[2] = WRD

		_trenutno++; // 0 --> 1 --> 2 --> 3

	}

};
